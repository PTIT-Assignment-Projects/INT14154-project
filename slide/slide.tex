\section{Cơ sở lý thuyết}

\subsection{LSTM}
\begin{frame}{Long Short-Term Memory (LSTM)}
\footnotesize
\textbf{Vấn đề của RNN truyền thống:} Vanishing/Exploding Gradient khi xử lý chuỗi dài.

\vspace{0.3cm}
\textbf{Giải pháp:} LSTM sử dụng 3 cổng (gates) để điều khiển luồng thông tin:
\begin{itemize}
    \item \textbf{Forget gate ($f_t$):} Quyết định thông tin nào giữ lại từ $c_{t-1}$
    \item \textbf{Input gate ($i_t$):} Quyết định thông tin mới nào lưu vào cell state
    \item \textbf{Output gate ($o_t$):} Quyết định thông tin nào xuất ra hidden state
\end{itemize}
\vspace{0.3cm}
\begin{align*}
f_t &= \sigma(W_{hf} h_{t-1} + W_{xf} x_t + b_f) \\
i_t &= \sigma(W_{hi} h_{t-1} + W_{xi} x_t + b_i) \\
\tilde{c}_t &= \tanh(W_{hc} h_{t-1} + W_{xc} x_t + b_c) \\
c_t &= f_t \odot c_{t-1} + i_t \odot \tilde{c}_t \\
o_t &= \sigma(W_{ho} h_{t-1} + W_{xo} x_t + b_o) \\
h_t &= o_t \odot \tanh(c_t)
\end{align*}
\end{frame}

\subsection{BiLSTM}
\begin{frame}{Bidirectional LSTM (BiLSTM)}
\textbf{Ý tưởng:} Xử lý chuỗi theo \textbf{cả hai hướng} để nắm bắt ngữ cảnh quá khứ và tương lai.

\vspace{0.3cm}
\begin{itemize}
    \item \textbf{Forward LSTM:} Đọc từ trái $\rightarrow$ phải, nắm bắt ngữ cảnh trước
    \item \textbf{Backward LSTM:} Đọc từ phải $\rightarrow$ trái, nắm bắt ngữ cảnh sau
    \item \textbf{Hidden state cuối:} Nối (concat) hai hướng
\end{itemize}
\vspace{0.3cm}
\[
\overrightarrow{h_t} = \text{LSTM}_{\rightarrow}(x_t, \overrightarrow{h_{t-1}}), \quad
\overleftarrow{h_t} = \text{LSTM}_{\leftarrow}(x_t, \overleftarrow{h_{t+1}})
\]
\[
h_t = [\overrightarrow{h_t} ; \overleftarrow{h_t}]
\]
\vspace{0.2cm}
\textbf{Ưu điểm:} Hiểu ngữ cảnh toàn cục tốt hơn so với LSTM đơn hướng.
\end{frame}

\subsection{GRU}
\begin{frame}{Gated Recurrent Unit (GRU)}
\textbf{Đặc điểm:} Biến thể đơn giản hóa của LSTM, chỉ có \textbf{2 cổng}:
\begin{itemize}
    \item \textbf{Update gate ($z_t$):} Kết hợp vai trò của forget + input gate
    \item \textbf{Reset gate ($r_t$):} Kiểm soát thông tin quá khứ cần "quên"
\end{itemize}
\vspace{0.3cm}
\begin{align*}
z_t &= \sigma(W_{hz} h_{t-1} + W_{xz} x_t + b_z) \\
r_t &= \sigma(W_{hr} h_{t-1} + W_{xr} x_t + b_r) \\
\tilde{h}_t &= \tanh(W_{hn}(r_t \odot h_{t-1}) + W_{xn} x_t + b_n) \\
h_t &= (1 - z_t) \odot \tilde{h}_t + z_t \odot h_{t-1}
\end{align*}
\vspace{0.2cm}
\textbf{Ưu điểm:} Ít tham số hơn LSTM $\rightarrow$ huấn luyện nhanh hơn, ít nguy cơ overfitting.
\end{frame}

\subsection{RCNN}
\begin{frame}{Recurrent Convolutional Neural Network (RCNN)}
\footnotesize
\textbf{Ý tưởng:} Kết hợp ưu điểm của RNN (ngữ cảnh tuần tự) và CNN (trích xuất đặc trưng cục bộ).

\vspace{0.3cm}
\textbf{Kiến trúc:}
\begin{enumerate}
    \item \textbf{Forward + Backward RNN:} Thu thập ngữ cảnh trái và phải cho mỗi từ
    \item \textbf{Concatenate:} Nối [ngữ cảnh trái + embedding từ + ngữ cảnh phải]
    \item \textbf{Fusion layer:} $\tanh(W \cdot [c_L; x; c_R] + b)$ -- tích hợp đặc trưng
    \item \textbf{Max pooling:} Lấy đặc trưng quan trọng nhất trên toàn bộ chuỗi
    \item \textbf{Fully connected:} Phân loại đầu ra
\end{enumerate}
\vspace{0.2cm}
\textbf{Ưu điểm:} Max pooling giúp tập trung vào các từ khóa quan trọng nhất, bất kể vị trí.
\end{frame}

\subsection{Transformer}
\begin{frame}{Transformer Encoder (Self-Attention)}
\textbf{Khác biệt cốt lõi:} Không dùng RNN $\rightarrow$ xử lý song song toàn bộ chuỗi.

\vspace{0.3cm}
\textbf{Self-Attention:} Mỗi từ "chú ý" đến tất cả từ khác trong câu:
\[
\text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^T}{\sqrt{d_k}}\right)V
\]
\vspace{0.2cm}
\textbf{Multi-Head Attention:} Chạy nhiều attention song song với các không gian con khác nhau:
\[
\text{MultiHead}(Q, K, V) = \text{Concat}(\text{head}_1, \ldots, \text{head}_h)W^O
\]
\vspace{0.2cm}
\textbf{Positional Encoding:} Bổ sung thông tin vị trí vì không có thứ tự tự nhiên:
\[
PE_{(pos, 2i)} = \sin\left(\frac{pos}{10000^{2i/d}}\right), \quad
PE_{(pos, 2i+1)} = \cos\left(\frac{pos}{10000^{2i/d}}\right)
\]
\end{frame}

\begin{frame}{Kiến trúc Transformer Encoder Block}
Mỗi block gồm 2 thành phần chính:
\begin{enumerate}
    \item \textbf{Multi-Head Self-Attention:}
    \begin{itemize}
        \item Tính attention giữa tất cả cặp từ
        \item Masking với attention mask (bỏ qua padding tokens)
    \end{itemize}
    \item \textbf{Feed-Forward Network:}
    \begin{itemize}
        \item 2 lớp tuyến tính: Linear $\rightarrow$ ReLU $\rightarrow$ Dropout $\rightarrow$ Linear
        \item Áp dụng độc lập cho từng vị trí
    \end{itemize}
\end{enumerate}
\vspace{0.2cm}
\textbf{Residual connection + Layer Normalization:}
\[
x' = \text{LayerNorm}(x + \text{Dropout}(\text{Attention}(x)))
\]
\[
x'' = \text{LayerNorm}(x' + \text{Dropout}(\text{FFN}(x')))
\]
\end{frame}

\subsection{Attention}
\begin{frame}{Cơ chế Attention}
\footnotesize
\textbf{Ý tưởng:} Thay vì dùng hidden state cuối cùng hoặc mean pooling, gán \textbf{trọng số khác nhau} cho từng time step.

\vspace{0.3cm}
\textbf{Self-Attention (Bahdanau-style):}
\begin{align*}
e_t &= \tanh(W_a h_t + b_a) \\
\alpha_t &= \text{softmax}(v^T e_t) \\
c &= \sum_{t=1}^{T} \alpha_t h_t
\end{align*}
\vspace{0.2cm}
\begin{itemize}
    \item $h_t$: hidden state tại time step $t$
    \item $\alpha_t$: trọng số attention (tổng = 1)
    \item $c$: context vector -- trọng số của toàn bộ chuỗi
\end{itemize}
\vspace{0.2cm}
\textbf{Ưu điểm:} Mô hình tự động học được từ/cụm từ nào quan trọng nhất cho việc phân loại.
\end{frame}

\section{Phương pháp tối ưu}

\begin{frame}{Class-Weighted Loss}
\textbf{Vấn đề:} Dữ liệu mất cân bằng nghiêm trọng (threat chỉ ~0.2\%, toxic ~9.4\%).

\vspace{0.3cm}
\textbf{Giải pháp:} BCEWithLogitsLoss với trọng số lớp:
\[
\text{pos\_weight}_c = \frac{\text{số mẫu âm}_c}{\text{số mẫu dương}_c}
\]
\[
\mathcal{L} = -\frac{1}{N} \sum_{i=1}^{N} \sum_{c=1}^{C} w_c \cdot [y_{ic} \log(\sigma(x_{ic})) + (1-y_{ic}) \log(1-\sigma(x_{ic}))]
\]
\vspace{0.2cm}
\textbf{Clip trọng số:} Giới hạn tối đa ở \textbf{50x} để tránh over-penalizing lớp đa số, cải thiện precision.
\end{frame}

\begin{frame}{AdamW Optimizer}
\textbf{AdamW} = Adam + \textbf{Weight Decay tách biệt} (decoupled).

\vspace{0.3cm}
\begin{itemize}
    \item \textbf{Adam thường:} Weight decay ghép vào gradient $\rightarrow$ không hiệu quả với adaptive learning rate
    \item \textbf{AdamW:} Áp dụng weight decay \textbf{trực tiếp} lên tham số sau bước cập nhật
\end{itemize}
\vspace{0.3cm}
\[
\theta_{t+1} = \theta_t - \eta \cdot \frac{\hat{m}_t}{\sqrt{\hat{v}_t} + \epsilon} - \eta \cdot \lambda \cdot \theta_t
\]
\vspace{0.2cm}
\begin{itemize}
    \item $\eta$: learning rate ($2 \times 10^{-3}$)
    \item $\lambda$: weight decay ($10^{-4}$)
    \item $\hat{m}_t, \hat{v}_t$: bias-corrected first/second moment
\end{itemize}
\end{frame}

\begin{frame}{Gradient Clipping}
\textbf{Vấn đề:} Exploding gradient -- gradient quá lớn làm cập nhật tham số mất ổn định.

\vspace{0.3cm}
\textbf{Giải pháp:} Cắt gradient nếu chuẩn vượt ngưỡng:
\[
\text{Nếu } \|\mathbf{g}\|_2 > \text{max\_norm} \Rightarrow \mathbf{g} \leftarrow \mathbf{g} \cdot \frac{\text{max\_norm}}{\|\mathbf{g}\|_2}
\]
\vspace{0.2cm}
\begin{itemize}
    \item \textbf{max\_norm} = 1.0
    \item Áp dụng trước \texttt{optimizer.step()}
    \item Đặc biệt quan trọng với RNN/LSTM (chuỗi dài)
\end{itemize}
\end{frame}

\begin{frame}{Warmup Cosine LR Scheduler}
\textbf{Warmup phase:} Tăng learning rate tuyến tính từ 0 $\rightarrow$ LR tối đa trong \textbf{10\%} tổng số bước.

\vspace{0.3cm}
\textbf{Cosine decay phase:} Giảm learning rate theo hàm cosine:
\[
\text{LR}(step) = \text{LR}_{\min} + (\text{LR}_{\max} - \text{LR}_{\min}) \cdot \frac{1}{2}\left(1 + \cos\left(\pi \cdot \frac{step - warmup}{total - warmup}\right)\right)
\]
\vspace{0.2cm}
\begin{itemize}
    \item \textbf{Warmup:} Tránh cập nhật quá mạnh ở đầu huấn luyện
    \item \textbf{Cosine decay:} Giảm dần mượt mà $\rightarrow$ hội tụ tốt hơn
\end{itemize}
\end{frame}

\begin{frame}{Early Stopping}
\textbf{Giám sát:} \textbf{Validation ROC-AUC (macro)} -- metric chính của bài toán.

\vspace{0.3cm}
\begin{itemize}
    \item \textbf{Patience:} 2 epochs không cải thiện $\rightarrow$ dừng
    \item \textbf{Mode:} Maximize (theo dõi ROC-AUC, không phải loss)
    \item \textbf{Checkpoint:} Lưu lại state của mô hình tốt nhất, restore khi dừng
\end{itemize}
\vspace{0.2cm}
\textbf{Tại sao dùng AUC thay vì loss?}
\begin{itemize}
    \item Class-weighted loss bị ảnh hưởng bởi trọng số $\rightarrow$ không phản ánh đúng chất lượng
    \item ROC-AUC đo khả năng phân biệt thực sự, bất kể threshold
\end{itemize}
\end{frame}

\begin{frame}{Optimal Threshold Search}
\textbf{Vấn đề:} Threshold mặc định 0.5 không tối ưu cho dữ liệu mất cân bằng.

\vspace{0.3cm}
\textbf{Giải pháp:} Tìm threshold tối ưu \textbf{cho từng lớp} riêng biệt:
\begin{itemize}
    \item Duyệt threshold từ 0.1 đến 0.9 (bước 0.01)
    \item Chọn threshold cho F1-score cao nhất trên validation set
    \item Mỗi lớp có threshold khác nhau phù hợp với phân phối riêng
\end{itemize}
\vspace{0.2cm}
\textbf{Kết quả:} Cải thiện đáng kể precision (giảm false positives) trong khi vẫn giữ recall cao.
\end{frame}

\end{itemize}
\end{frame}

\begin{frame}{Hướng phát triển}
\begin{itemize}
    \item \textbf{Pretrained embeddings:} GloVe, FastText thay vì embedding ngẫu nhiên
    \item \textbf{Fine-tune class weights:} Tìm trọng số tối ưu thay vì ratio đơn thuần + clip
    \item \textbf{Ensemble:} Kết hợp predictions từ Attention BiLSTM + RCNN
    \item \textbf{Data augmentation:} Back-translation, synonym replacement cho lớp thiểu số
    \item \textbf{Hyperparameter tuning:} Grid search / Bayesian optimization cho từng mô hình
    \item \textbf{Transformer tuning:} Tăng số epoch, giảm learning rate, tăng số head
\end{itemize}
\end{frame}

\begin{frame}{Kết luận}
\begin{itemize}
    \item Đã cài đặt từ gốc \textbf{7 mô hình} deep learning cho phân loại văn bản
    \item \textbf{RCNN} đạt ROC-AUC cao nhất (0.984), \textbf{Attention BiLSTM} có F1 Macro tốt nhất (0.63)
    \item Full data (159K) cải thiện vượt trội so với 10K mẫu: AUC +0.02, F1 Macro +0.20
    \item Các phương pháp tối ưu (AdamW, Gradient Clipping, LR Scheduler, Class Weights, Optimal Thresholds) giúp mô hình hội tụ ổn định
    \item \textbf{Thách thức chính:} Mất cân bằng lớp nghiêm trọng $\rightarrow$ precision thấp ở các lớp thiểu số
    \item Cần pretrained embeddings và fine-tuning để cải thiện thêm
\end{itemize}
\end{frame}
